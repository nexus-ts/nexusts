# 28 · Inertia + React (SPA)

Inertia.js v3 with **React** in client-side-only mode. This is the
"Laravel + Inertia (CSR)" recipe: the server returns an Inertia page
object, the client mounts React and renders it.

## What it shows

- Inertia page rendering via `inertia.render(component, props)`
- 303 redirect from a POST action (`/counter` increments and redirects to `/`)
- 422 form validation errors via Inertia's `errors` prop
- Inertia `<a method="post">` link helper
- Form-submit with retained `old` input on error

## How to run

```bash
cd examples/28-inertia-react-spa
bun install                 # first time only — installs React peer-deps at the workspace root
bun build ./frontend/app.tsx --outdir=./public --target=browser --format=esm --minify
bun main.ts
```

Open <http://localhost:3000> and click `+1` or submit the form with
an empty name to see the validation error.

> **Note on bundling:** the first `bun build` step compiles the React
> client into `./public/app.js`. In a real app you'd run this in a
> `dev` script that watches for changes. See "Dev mode" below.

## Dev mode (with watch)

```bash
# Terminal 1: rebuild the client on every change
bun build ./frontend/app.tsx --outdir=./public --target=browser --format=esm --watch

# Terminal 2: run the server
bun main.ts
```

## Code

### Server — `main.ts`

```ts
import "reflect-metadata";
import path from "node:path";
import { Application, Module, Controller, Get, Post, Body, Inject, Ctx, Injectable } from "@nexusts/core";
import { StaticModule } from "@nexusts/static";
import { Inertia } from "@nexusts/view";

@Injectable()
@Controller("/")
class HomeController {
  @Inject(Inertia.TOKEN) declare private inertia: Inertia;

  @Get("/")
  home() {
    return this.inertia.render("Home", {
      greeting: "Hello from Inertia + React!",
      count: readCount(),
    });
  }

  @Post("/counter")
  counter() {
    bumpCount();
    return this.inertia.location("/");          // 303 See Other
  }

  @Post("/greet")
  greet(@Ctx() c: any, @Body() body: { name?: string }) {
    if (!body?.name || body.name.trim().length === 0) {
      c.status(422);
      return this.inertia.render("Home", {
        greeting: "Hello from Inertia + React!",
        count: readCount(),
        errors: { name: "Please tell us your name." },
      });
    }
    return this.inertia.render("Home", {
      greeting: `Welcome, ${body.name}!`,
      count: readCount(),
    });
  }
}

@Module({
  providers: [{ provide: Inertia.TOKEN, useValue: new Inertia() }],
  controllers: [HomeController],
})
class AppModule {}

const app = new Application(AppModule);

app.server.app.use("/static/*", StaticModule.mount({
  root: path.join(import.meta.dir, "public"),
  prefix: "/static",
}));

await app.listen(3000);
```

### Client — `frontend/app.tsx`

```tsx
import { createInertiaApp } from "@inertiajs/react";
import { createRoot } from "react-dom/client";
import { useState } from "react";

createInertiaApp({
  resolve: (name) => Home,            // Inertia sends `component: "Home"`
  setup({ el, App, props }) {
    createRoot(el).render(<App {...props} />);
  },
});

function Home({ greeting, count, errors }: any) {
  return (
    <main>
      <h1>{greeting}</h1>
      <p>Counter: {count}</p>
      <a href="/counter" method="post" as="button">+1</a>
      <form action="/greet" method="post">
        <input name="name" />
        {errors?.name && <p style={{ color: "crimson" }}>{errors.name}</p>}
        <button>Submit</button>
      </form>
    </main>
  );
}
```

## Smoke test note

The boot test (`tests/examples/smoke.test.ts`) only verifies the
**server** starts cleanly. The client bundle is **not** regenerated
by the smoke test — run the `bun build` step once after `bun install`
and you're set.
